   spatial_jepas: list[JEPALoss] | None = None,\n    temporal_jepas: list[JEPALoss] | None = None,\n    use_pooled_vicreg: bool = False,\n) -> dict:\n    """\n    Train one epoch with the specified objective.\n\n    Returns dict with losses and metrics.\n    """\n    n_samples = train_grid.shape[0]\n    perm = rng.permutation(n_samples)\n\n    total_obj_loss = 0.0\n    total_pooled_var = 0.0\n    total_pooled_cov = 0.0\n    total_pooled_std = 0.0\n    n_batches = 0\n\n    for start in range(0, n_samples, batch_size):\n        end = min(start + batch_size, n_samples)\n        batch = train_grid[perm[start:end]]\n\n        fwd = encoder.forward_with_intermediates(batch)\n\n        # =====================================================================\n        #  Objective-specific code-gradient computation\n        # =====================================================================\n\n        if objective == "jepa":\n            # ---- Spatial JEPA ----\n            spatial_code_grads = []\n            spatial_losses = []\n            for l in range(encoder.n_spatial_layers):\n                out = fwd["spatial_outputs"][l]         # (B, S_l, T, d)\n                B, S_l, T, d = out.shape\n                jepa_in = reshape_for_spatial_jepa(out) # (B*T, S_l, d)\n                result = spatial_jepas[l].step([jepa_in])\n                spatial_losses.append(result["loss"])\n                cg = result["code_grads"][0]\n                spatial_code_grads.append(reshape_spatial_grads_back(cg, B, S_l, T, d))\n\n            # ---- Temporal JEPA ----\n            temporal_code_grads = []\n            temporal_losses = []\n            for l in range(encoder.n_temporal_layers):\n                out = fwd["temporal_outputs"][l]         # (B, T_l, S_f, d)\n                B, T_l, S_f, d = out.shape\n                jepa_in = reshape_for_temporal_jepa(out) # (B*S_f, T_l, d)\n                result = temporal_jepas[l].step([jepa_in])\n                temporal_losses.append(result["loss"])\n                cg = result["code_grads"][0]\n                temporal_code_grads.append(reshape_temporal_grads_back(cg, B, T_l, S_f, d))\n\n            obj_loss = ALPHA * np.mean(spatial_losses) + (1.0 - ALPHA) * np.mean(temporal_losses)\n\n        elif objective == "sfa":\n            # Slowness on FINAL temporal output only\n            z = fwd["temporal_outputs"][-1]  # (B, T_f, S_f, d_out)\n            sfa = sfa_loss_and_grad(z)\n\n            spatial_code_grads = [np.zeros_like(fwd["spatial_outputs"][l])\n                                   for l in range(encoder.n_spatial_layers)]\n            temporal_code_grads = [np.zeros_like(fwd["temporal_outputs"][l])\n                                    for l in range(encoder.n_temporal_layers)]\n            temporal_code_grads[-1] = sfa["grad"]\n\n            obj_loss = sfa["loss"]\n\n        elif objective == "hebbian":\n            # Variance maximisation on ALL intermediate codes\n            hebb_losses = []\n            spatial_code_grads = []\n            temporal_code_grads = []\n\n            for l in range(encoder.n_spatial_layers):\n                z = fwd["spatial_outputs"][l]\n                h = hebbian_loss_and_grad(z, LAMBDA_HEBB)\n                spatial_code_grads.append(h["grad"])\n                hebb_losses.append(h["loss"])\n\n            for l in range(encoder.n_temporal_layers):\n                z = fwd["temporal_outputs"][l]\n                h = hebbian_loss_and_grad(z, LAMBDA_HEBB)\n                temporal_code_grads.append(h["grad"])\n                hebb_losses.append(h["loss"])\n\n            obj_loss = np.mean(hebb_losses)\n\n        elif objective == "recon":\n            # Reconstruction + sparsity on ALL layers\n            recon_losses = []\n            spatial_code_grads = []\n            temporal_code_grads = []\n            dec_grads_spatial = []\n            dec_grads_temporal = []\n\n            node = encoder.master_spatial\n            W_dec = node.W_dec\n            b_dec = node.b_dec\n\n            for l in range(encoder.n_spatial_layers):\n                code = fwd["spatial_outputs"][l]\n                inp = fwd["spatial_inputs"][l]\n                r = recon_loss_and_grad(code, inp, W_dec, b_dec, LAMBDA_L1)\n                spatial_code_grads.append(r["code_grad"])\n                recon_losses.append(r["loss"])\n                dec_grads_spatial.append({"W_dec": r["W_dec_grad"], "b_dec": r["b_dec_grad"]})\n\n            for l in range(encoder.n_temporal_layers):\n                code = fwd["temporal_outputs"][l]\n                inp = fwd["temporal_inputs"][l]\n                r = recon_loss_and_grad(code, inp, W_dec, b_dec, LAMBDA_L1)\n                temporal_code_grads.append(r["code_grad"])\n                recon_losses.append(r["loss"])\n                dec_grads_temporal.append({"W_dec": r["W_dec_grad"], "b_dec": r["b_dec_grad"]})\n\n            obj_loss = np.mean(recon_losses)\n\n            # Update decoder separately (encoder.backward() gives zero decoder grads)\n            if adam_decoder is not None:\n                avg_W_dec_grad = (\n                    sum(g["W_dec"] for g in dec_grads_spatial + dec_grads_temporal)\n                    / (encoder.n_spatial_layers + encoder.n_temporal_layers)\n                )\n                avg_b_dec_grad = (\n                    sum(g["b_dec"] for g in dec_grads_spatial + dec_grads_temporal)\n                    / (encoder.n_spatial_layers + encoder.n_temporal_layers)\n                )\n                adam_decoder.step(\n                    {"W_dec": W_dec, "b_dec": b_dec},\n                    {"W_dec": avg_W_dec_grad, "b_dec": avg_b_dec_grad},\n                )\n\n        else:\n            raise ValueError(f"Unknown objective: {objective}")\n\n        # =====================================================================\n        #  Pooled VICReg injection (same mechanism as Phase 3)\n        # =====================================================================\n        if use_pooled_vicreg:\n            z_pooled = fwd["pooled"]  # (B, d_out)\n            dL_dzp = pooled_vicreg_grad(z_pooled)  # (B, d_out)\n\n            T_final, S_final = fwd["temporal_outputs"][-1].shape[1:3]\n            dL_expanded = dL_dzp[:, None, None, :] / (T_final * S_final)\n            dL_expanded = np.broadcast_to(\n                dL_expanded, fwd["temporal_outputs"][-1].shape\n            ).copy()\n\n            # Scale by 1/(1−α) to match temporal backward weighting\n            temporal_code_grads[-1] = temporal_code_grads[-1] + dL_expanded / (1.0 - ALPHA)\n\n            pv = pooled_vicreg_loss(z_pooled)\n            total_pooled_var += pv["var_loss"]\n            total_pooled_cov += pv["cov_loss"]\n            total_pooled_std += pv["std"]\n        else:\n            total_pooled_std += float(\n                np.sqrt(fwd["pooled"].var(axis=0, ddof=0).mean() + 1e-12)\n            )\n\n        # =====================================================================\n        #  Encoder backward + update\n        # =====================================================================\n        if adam_spatial is not None or adam_temporal is not None:\n            grads = encoder.backward(\n                fwd,\n                dL_dspatial_codes=spatial_code_grads,\n                dL_dtemporal_codes=temporal_code_grads,\n                alpha=ALPHA,\n            )\n\n            if encoder.variant == "P3-C":\n                combined = {\n                    k: grads["dL_dspatial"][k] + grads["dL_dtemporal"][k]\n                    for k in grads["dL_dspatial"]\n                }\n                adam_spatial.step(\n                    {\n                        "W_enc": encoder.master_spatial.W_enc,\n                        "b_enc": encoder.master_spatial.b_enc,\n                        "W_dec": encoder.master_spatial.W_dec,\n                        "b_dec": encoder.master_spatial.b_dec,\n                    },\n                    combined,\n                )\n            else:\n                if adam_spatial is not None:\n                    adam_spatial.step(\n                        {\n                            "W_enc": encoder.master_spatial.W_enc,\n                            "b_enc": encoder.master_spatial.b_enc,\n                            "W_dec": encoder.master_spatial.W_dec,\n                            "b_dec": encoder.master_spatial.b_dec,\n                        },\n                        grads["dL_dspatial"],\n                    )\n                if adam_temporal is not None:\n                    adam_temporal.step(\n                        {\n                            "W_enc": encoder.master_temporal.W_enc,\n                            "b_enc": encoder.master_temporal.b_enc,\n                            "W_dec": encoder.master_temporal.W_dec,\n                            "b_dec": encoder.master_temporal.b_dec,\n                        },\n                        grads["dL_dtemporal"],\n                    )\n\n        total_obj_loss += obj_loss\n        n_batches += 1\n\n    return {\n        "obj_loss": total_obj_loss / n_batches,\n        "pooled_var": total_pooled_var / n_batches,\n        "pooled_cov": total_pooled_cov / n_batches,\n        "pooled_std": total_pooled_std / n_batches,\n    }\n\n\n# =============================================================================\n#  4. Classification evaluation\n# =============================================================================\n\ndef evaluate_classification(\n    encoder: SpatiotemporalEncoder,\n    train_grid: np.ndarray,\n    train_y: np.ndarray,\n    test_grid: np.ndarray,\n    test_y: np.ndarray,\n    seed: int,\n) -> dict:\n    """\n    Fit SimpleLogisticRegression on spatial_pooled_then_flat readout.\n    """\n    fwd_train = encoder.forward_with_intermediates(train_grid)\n    fwd_test = encoder.forward_with_intermediates(test_grid)\n\n    # spatial_pooled_then_flat: mean over spatial axis, flatten\n    tr = fwd_train["temporal_outputs"][-1].mean(axis=2).reshape(len(train_y), -1)\n    te = fwd_test["temporal_outputs"][-1].mean(axis=2).reshape(len(test_y), -1)\n\n    probe = SimpleLogisticRegression(\n        n_classes=N_CLASSES,\n        n_features=tr.shape[1],\n        lr=0.1,\n        max_iter=2000,\n        seed=seed,\n    )\n    probe.fit(tr, train_y)\n\n    train_acc = float(probe.score(tr, train_y))\n    test_acc = float(probe.score(te, test_y))\n\n    per_class = {}\n    for cls in range(N_CLASSES):\n        mask = test_y == cls\n        per_class[f"class_{cls}_acc"] = float(probe.score(te[mask], test_y[mask])) if mask.sum() > 0 else 0.0\n\n    return {\n        "train_acc": train_acc,\n        "test_acc": test_acc,\n        **per_class,\n    }\n\n\n# =============================================================================\n#  5. Single experiment runner\n# =============================================================================\n\ndef run_single_experiment(args: tuple) -> dict:\n    """\n    args = (objective, seed, use_pooled_vicreg, epochs)\n\n    objective: \'jepa\' | \'sfa\' | \'hebbian\' | \'recon\' | \'untrained\'\n    """\n    objective, seed, use_pooled_vicreg, epochs = args\n\n    rng = np.random.default_rng(seed)\n\n    ds = generate_spatiotemporal_dataset(\n        n_train_per_class=N_TRAIN_PER_CLASS,\n        n_test_per_class=N_TEST_PER_CLASS,\n        noise_flip_prob=0.10,\n        seed=seed,\n    )\n    train_grid = ds["train_grid"]\n    train_y = ds["train_y"]\n    test_grid = ds["test_grid"]\n    test_y = ds["test_y"]\n\n    encoder = SpatiotemporalEncoder(\n        variant="P3-C",\n        d=D,\n        d_out=D_OUT,\n        n_spatial_layers=3,\n        n_temporal_layers=3,\n        l1_lambda=0.0,\n        seed=seed,\n    )\n\n    update_encoder = (objective != "untrained")\n\n    # JEPA losses (only used by jepa objective)\n    spatial_jepas = None\n    temporal_jepas = None\n    if objective == "jepa" and update_encoder:\n        spatial_jepas = create_jepa_losses(encoder.n_spatial_layers, D_OUT, lr=LR)\n        temporal_jepas = create_jepa_losses(encoder.n_temporal_layers, D_OUT, lr=LR)\n\n    # Adam optimisers\n    adam_spatial = None\n    adam_temporal = None\n    adam_decoder = None\n    if update_encoder:\n        params_enc = {\n            "W_enc": encoder.master_spatial.W_enc,\n            "b_enc": encoder.master_spatial.b_enc,\n            "W_dec": encoder.master_spatial.W_dec,\n            "b_dec": encoder.master_spatial.b_dec,\n        }\n        adam_spatial = _Adam(params_enc, lr=LR)\n        adam_temporal = adam_spatial  # P3-C: shared\n\n        if objective == "recon":\n            adam_decoder = _Adam(\n                {"W_dec": encoder.master_spatial.W_dec,\n                 "b_dec": encoder.master_spatial.b_dec},\n                lr=LR,\n            )\n\n    # Training\n    t0 = time.time()\n    for epoch in range(epochs):\n        metrics = train_epoch(\n            encoder,\n            train_grid,\n            BATCH_SIZE,\n            rng,\n            objective=objective,\n            adam_spatial=adam_spatial if update_encoder else None,\n            adam_temporal=adam_temporal if update_encoder else None,\n            adam_decoder=adam_decoder if update_encoder else None,\n            spatial_jepas=spatial_jepas,\n            temporal_jepas=temporal_jepas,\n            use_pooled_vicreg=use_pooled_vicreg,\n        )\n    t1 = time.time()\n\n    # Final pooled std\n    final_fwd = encoder.forward_with_intermediates(train_grid)\n    final_pooled_std = float(\n        np.sqrt(final_fwd["pooled"].var(axis=0, ddof=0).mean() + 1e-12)\n    )\n\n    # Classification\n    eval_res = evaluate_classification(encoder, train_grid, train_y,\n                                       test_grid, test_y, seed=seed)\n\n    return {\n        "objective": objective,\n        "seed": seed,\n        "use_pooled_vicreg": use_pooled_vicreg,\n        "train_acc": eval_res["train_acc"],\n        "test_acc": eval_res["test_acc"],\n        "class_0_acc": eval_res["class_0_acc"],\n        "class_1_acc": eval_res["class_1_acc"],\n        "class_2_acc": eval_res["class_2_acc"],\n        "class_3_acc": eval_res["class_3_acc"],\n        "final_loss": metrics["obj_loss"],\n        "final_pooled_std": final_pooled_std,\n        "training_time_sec": t1 - t0,\n    }\n\n\n# =============================================================================\n#  6. CSV writer\n# =============================================================================\n\nFIELDNAMES = [\n    "objective", "seed", "use_pooled_vicreg", "train_acc", "test_acc",\n    "class_0_acc", "class_1_acc", "class_2_acc", "class_3_acc",\n    "final_loss", "final_pooled_std", "training_time_sec",\n]\n\n\ndef save_result(result: dict, csv_path: str) -> None:\n    file_exists = os.path.exists(csv_path)\n    with open(csv_path, "a", newline="") as f:\n        writer = csv.DictWriter(f, fieldnames=FIELDNAMES)\n        if not file_exists:\n            writer.writeheader()\n        writer.writerow(result)\n\n\n# =============================================================================\n#  7. Main experiment orchestration\n# =============================================================================\n\ndef main(dry_run: bool = False):\n    os.makedirs(OUTPUT_DIR, exist_ok=True)\n\n    if os.path.exists(RESULTS_CSV):\n        os.remove(RESULTS_CSV)\n\n    # Build task list\n    tasks = []\n    task_labels = []\n    for objective in OBJECTIVES:\n        for use_pv in [False, True]:\n            if dry_run:\n                seeds_to_use = [42]\n                ep = 1\n            else:\n                seeds_to_use = SEEDS\n                ep = EPOCHS\n            for seed in seeds_to_use:\n                label = f"{objective.upper()}{\'+\' if use_pv else \'-\'}VICReg_seed{seed}"\n                tasks.append((objective, seed, use_pv, ep))\n                task_labels.append(label)\n\n    # Untrained baseline\n    if dry_run:\n        tasks.append(("untrained", 42, False, 1))\n        task_labels.append("UNTRAINED_seed42")\n    else:\n        for seed in SEEDS:\n            tasks.append(("untrained", seed, False, EPOCHS))\n            task_labels.append(f"UNTRAINED_seed{seed}")\n\n    total_tasks = len(tasks)\n    mode = "DRY-RUN" if dry_run else "FULL RUN"\n    print("=" * 70)\n    print(f"  PHASE 4 — Objective Comparison — {mode}")\n    print(f"  Tasks: {total_tasks}")\n    print(f"  Epochs: {ep if dry_run else EPOCHS}, Batch: {BATCH_SIZE}, LR: {LR}")\n    print(f"  Seeds: {SEEDS if not dry_run else [42]}")\n    print("=" * 70)\n\n    t_start = time.time()\n\n    if dry_run:\n        all_results = [run_single_experiment(t) for t in tasks]\n    else:\n        n_workers = min(5, total_tasks)\n        with Pool(processes=n_workers) as pool:\n            all_results = pool.map(run_single_experiment, tasks)\n\n    t_end = time.time()\n\n    for r in all_results:\n        save_result(r, RESULTS_CSV)\n\n    print(f"\\n  All runs completed in {t_end - t_start:.1f}s")\n    print(f"  Results saved to {RESULTS_CSV}")\n\n    # Quick summary\n    print("\\n  Summary (mean ± std test accuracy):")\n    for objective in OBJECTIVES + ["untrained"]:\n        for use_pv in ([False, True] if objective != "untrained" else [False]):\n            rows = [r for r in all_results\n                    if r["objective"] == objective and r["use_pooled_vicreg"] == use_pv]\n            if rows:\n                accs = [r["test_acc"] for r in rows]\n                mean_acc = np.mean(accs)\n                std_acc = np.std(accs, ddof=1 if len(accs) > 1 else 0)\n                label = f"{objective.upper()}{\'+VICReg\' if use_pv else \'\'}"\n                print(f"    {label:22s}: {mean_acc:.4f} ± {std_acc:.4f}")\n\n    print("=" * 70)\n    return all_results\n\n\n# =============================================================================\n#  8. Statistical analysis & report\n# =============================================================================\n\ndef run_statistical_analysis(results: list[dict]) -> None:\n    """Generate comprehensive markdown report."""\n    try:\n        from scipy import stats\n    except ImportError:\n        print("  scipy not available — skipping statistical tests.")\n        stats = None\n\n    # Organise results\n    grouped: dict[tuple, list[dict]] = {}\n    for r in results:\n        key = (r["objective"], r["use_pooled_vicreg"])\n        grouped.setdefault(key, []).append(r)\n\n    # Untrained baseline\n    untrained_rows = grouped.get(("untrained", False), [])\n    untrained_accs = np.array([r["test_acc"] for r in untrained_rows])\n\n    # ---- Summary table ----\n    summary: dict[tuple, dict] = {}\n    for key, rows in grouped.items():\n        accs = np.array([r["test_acc"] for r in rows])\n        times = np.array([r["training_time_sec"] for r in rows])\n        stds = np.array([r["final_pooled_std"] for r in rows])\n        summary[key] = {\n            "mean_acc": float(accs.mean()),\n            "std_acc": float(np.std(accs, ddof=1)) if len(accs) > 1 else 0.0,\n            "mean_time": float(times.mean()),\n            "mean_std": float(stds.mean()),\n            "accs": accs,\n            "rows": rows,\n        }\n\n    # ---- Paired t-tests vs untrained ----\n    ttest_results: dict[tuple, dict] = {}\n    if stats is not None and len(untrained_accs) > 0:\n        for key, s in summary.items():\n            if key[0] == "untrained":\n                continue\n            accs = s["accs"]\n            # Pair by seed\n            paired_untrained = []\n            paired_obj = []\n            for r in s["rows"]:\n                for u in untrained_rows:\n                    if r["seed"] == u["seed"]:\n                        paired_obj.append(r["test_acc"])\n                        paired_untrained.append(u["test_acc"])\n                        break\n            if len(paired_obj) > 1:\n                t_stat, p_val = stats.ttest_rel(paired_obj, paired_untrained)\n                gain = float(np.mean(paired_obj) - np.mean(paired_untrained))\n                diff = np.array(paired_obj) - np.array(paired_untrained)\n                cohens_d = float(diff.mean() / (diff.std(ddof=1) + 1e-12))\n                ttest_results[key] = {\n                    "gain": gain,\n                    "t": t_stat,\n                    "p": p_val,\n                    "cohens_d": cohens_d,\n                }\n\n    # ---- Falsification criteria ----\n    jepa_pv_key = ("jepa", True)\n    jepa_mean = summary.get(jepa_pv_key, {}).get("mean_acc", 0.0)\n\n    f1_triggered = jepa_mean < 0.55\n\n    f2_triggered = False\n    f2_obj = None\n    for key, s in summary.items():\n        if key[0] == "untrained":\n            continue\n        if s["mean_acc"] > jepa_mean + 0.03:\n            f2_triggered = True\n            f2_obj = key\n\n    f3_results: dict[str, bool] = {}\n    for obj in OBJECTIVES:\n        with_key = summary.get((obj, True), {"mean_acc": 0.0})["mean_acc"]\n        without_key = summary.get((obj, False), {"mean_acc": 0.0})["mean_acc"]\n        f3_results[obj] = without_key >= with_key\n\n    # ---- Build report ----\n    lines = []\n    lines.append("# Phase 4 — Training Objective Comparison Report")\n    lines.append("")\n    lines.append(f"**Generated**: {time.strftime(\'%Y-%m-%d %H:%M:%S\')}")\n    lines.append("")\n\n    # Config\n    lines.append("## Experiment Configuration")\n    lines.append("")\n    lines.append(f"- **Architecture**: P3-C (d={D}, d_out={D_OUT})")\n    lines.append(f"- **Encoder layers**: 3 spatial + 3 temporal")\n    lines.append(f"- **Readout**: spatial_pooled_then_flat (26×16=416 dims)")\n    lines.append(f"- **Train/test**: {N_TRAIN_PER_CLASS} / {N_TEST_PER_CLASS} per class")\n    lines.append(f"- **Epochs**: {EPOCHS}, Batch size: {BATCH_SIZE}, LR: {LR}")\n    lines.append(f"- **Alpha**: {ALPHA}")\n    lines.append(f"- **Lambda Hebb**: {LAMBDA_HEBB}")\n    lines.append(f"- **Lambda L1 (recon)**: {LAMBDA_L1}")\n    lines.append(f"- **Seeds**: {SEEDS}")\n    lines.append("")\n\n    # Mathematical formulations\n    lines.append("## Mathematical Formulations")\n    lines.append("")\n    lines.append("### 1. JEPA")\n    lines.append("""\nFor each layer code `Z ∈ ℝ^{N×P×d}`:\n- Prediction: `ẑ_{p+1} = W_pred·z_p + b_pred`\n- `L_pred = ½·mean((ẑ_{p+1} − z_{p+1})²) + ½·mean((ẑ_p − z_p)²)`\n- `L_var = mean_j max(0, 1 − σ_j(Z))`\n- `L_cov = (1/d) Σ_{j≠k} C_{jk}²`\n- `L_JEPA = L_pred + 25·L_var + 25·L_cov`\n""")\n    lines.append("### 2. SFA (Slow Feature Analysis)")\n    lines.append("""\nOn final temporal output `z ∈ ℝ^{B×T×S×d}`:\n- `L_slow = (1/(B·(T−1)·S·d)) Σ_{b,s,t≥1} ‖z_{b,t,s} − z_{b,t−1,s}‖²`\n- With pooled VICReg: `L = L_slow + L_VICReg(z̄_pooled)` — standard gradient-based SFA (Lagrangian relaxation)\n- Without VICReg: pure slowness (expected collapse to constant)\n""")\n    lines.append("### 3.
