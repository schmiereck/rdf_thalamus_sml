def extract_features_batched(encoder, grid, batch_size=64):
    features = []
    for start in range(0, len(grid), batch_size):
        batch = grid[start:start+batch_size]
        fwd = encoder.forward_with_intermediates(batch)
        feat = fwd["temporal_outputs"][-1].mean(axis=2).reshape(len(batch), -1)
        features.append(feat)
    return np.concatenate(features, axis=0)
