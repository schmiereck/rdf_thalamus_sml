=============================================================================\n\ndef disentangle_loss_and_grad(\n    code: np.ndarray,\n    beta: float = BETA_DISENTANGLE,\n) -> dict:\n    """\n    Correlation penalty on code dimensions.\n    \n    L_dis = beta * (1/(d*(d-1))) * sum_{j!=k} Corr(code_j, code_k)^2\n    \n    Parameters\n    ----------\n    code : (B, P1, P2, d) intermediate code\n    beta : disentanglement weight\n    \n    Returns dict with \'loss\', \'code_grad\'\n    """\n    B, P1, P2, d = code.shape\n    M = B * P1 * P2\n    z = code.reshape(M, d)  # (M, d)\n    \n    # Compute correlation matrix\n    mu = z.mean(axis=0, keepdims=True)  # (1, d)\n    std = np.sqrt(z.var(axis=0, ddof=0, keepdims=True) + 1e-12)  # (1, d)\n    z_norm = (z - mu) / std  # (M, d) normalized\n    \n    C = (z_norm.T @ z_norm) / M  # (d, d) correlation matrix\n    mask = 1.0 - np.eye(d)\n    C_off = C * mask  # off-diagonal only\n    \n    loss = beta * np.sum(C_off ** 2) / (d * (d - 1))\n    \n    # Gradient: dL/dz\n    # dL/dC = beta * 2 * C_off / (d*(d-1))\n    # dC/dz_norm = 2 * z_norm / M (simplified)\n    # dL/dz = dL/dz_norm * dz_norm/dz\n    dL_dC = beta * 2.0 * C_off / (d * (d - 1))  # (d, d)\n    dL_dz_norm = (2.0 / M) * (z_norm @ dL_dC)  # (M, d)\n    \n    # dz_norm/dz = (1/std) - z_norm * (z_norm * dL_dz_norm).sum(1,keepdims)/(M*std)\n    # Simplified: dz/dz_norm = std, so dz_norm/dz = 1/std\n    # But we need dz/dL = dz/dz_norm * dz_norm/dL\n    # Actually: z_norm = (z - mu) / std, so dz_norm/dz = (I - 1/M) / std\n    # Full gradient: dL/dz = dL/dz_norm * dz_norm/dz\n    # dz_norm/dz per sample: (1/std) * (I - 1/M * 11^T) ≈ 1/std for large M\n    \n    # More precise gradient:\n    # dz_norm/dz_i = (1/std) * (e_i - mean_batch_grad/M)\n    # For simplicity with large M:\n    dL_dz = dL_dz_norm / std  # (M, d)\n    \n    code_grad = dL_dz.reshape(B, P1, P2, d)\n    return {"loss": float(loss), "code_grad": code_grad}\n\n\n#
